--Services
local Players = game:GetService("Players")
local Teams = game:GetService("Teams")
local LobbyTP = game.Workspace.TPLobby
local GameTP = game.Workspace.TPGame
--Timers
local Intermission = 10
local GameTime = 100
local SeekerTime = 10
--Minimum Players to Start
local MinPlayers = 4
local RE = game.ReplicatedStorage.RemoteEvent

game.Players.PlayerAdded:Connect(function(player)
	player.CharacterAdded:Connect(function(character)
		print(player.Name .. " has respawned!")
		player.Team = Teams.Spectator
	end)
end)

local function AssignTeams()
	local PList = Players:GetPlayers()
	local totalplayers = #Players:GetPlayers()

	if totalplayers < 4 then
		local randomPlayer = PList[math.random(1, totalplayers)]
		randomPlayer.Team = Teams.Seeker

	elseif totalplayers >= 4 and totalplayers < 8 then
		local randomPlayer = PList[math.random(1, totalplayers)]
		randomPlayer.Team = Teams.Seeker
		local lastseeker = randomPlayer

		local randomPlayer = PList[math.random(1, totalplayers)]
		while randomPlayer == lastseeker do
			randomPlayer = PList[math.random(1, totalplayers)]
		end
		randomPlayer.Team = Teams.Seeker

	elseif totalplayers > 8 then
		local randomPlayer = PList[math.random(1, totalplayers)]
		randomPlayer.Team = Teams.Seeker
		local lastseeker = randomPlayer

		local randomPlayer = PList[math.random(1, totalplayers)]
		while randomPlayer == lastseeker do
			randomPlayer = PList[math.random(1, totalplayers)]
		end
		randomPlayer.Team = Teams.Seeker
		local lastseeker = randomPlayer

		local randomPlayer = PList[math.random(1, totalplayers)]
		while randomPlayer == lastseeker do
			randomPlayer = PList[math.random(1, totalplayers)]
		end
		randomPlayer.Team = Teams.Seeker
	end


	for i, player in ipairs(Players:GetPlayers()) do
		if player.Team ~= Teams.Seeker then
			player.Team = Teams.Hider
		end
	end	
end

local function TpToGame()
	local pls = Players:GetPlayers()
	for i, player in pls do
		if player.Team == Teams.Hider then
			local character = player.Character
			character:PivotTo(GameTP:GetPivot())
		end	
	end
	for count = SeekerTime, 0, -1 do
		task.wait(1)
		print(count)	
	end 
	for i, player in pls do		
		if player.Team == Teams.Seeker then
			local character = player.Character
			character:PivotTo(GameTP:GetPivot())
		end
	end
end

local function GiveWeapon()
	local pls = Players:GetPlayers()
	-- go to players inventory and clone gun that is in replicated storage and give to player
	for i, player in pls do
		if player.Team == Teams.Seeker then
			local gun = game.ReplicatedStorage.Gun:Clone()
			gun.Parent = player.Backpack
		end
	end
end

local function ClearWeapons()
	local pls = Players:GetPlayers()
	print("Removing weapons")
	task.wait(1)
	-- go to players inventory and remove gun/clear backpack and items
	for i, player in pls do
		print(player.Name)
		player:WaitForChild("Backpack"):ClearAllChildren()
		for _, item in pairs(player.Character:GetChildren()) do
			if item:IsA("Tool") then
				item:Destroy()
			end
		end
	end
end

local function ClearTeams()
	local pls = Players:GetPlayers()
	for i, player in pls do
		player.Team = Teams.Spectator
	end
end

local function runGame()
	local pls = Players:GetPlayers()
	local Players = game:GetService("Players")
	local Teams = game:GetService("Teams")
	for count = GameTime, 0, -1 do
		local seekers = #Teams.Seeker:GetPlayers()
		local hiders = #Teams.Hider:GetPlayers()
		if seekers == 0 and hiders == 0 then
			print("It's a draw!")
			RE:FireAllClients("It's a draw!")
			task.wait(5)
			break
		elseif seekers == 0 then
			print("Hider team wins!")
			RE:FireAllClients("Hider team wins!")
			task.wait(5)
			break
		elseif hiders == 0 then
			print("Seeker team wins!")
			RE:FireAllClients("Seeker team wins!")
			task.wait(5)
			break
		end
		print(count)
		task.wait(1)
	end
end 

local function TpToLobby()
	local pls = Players:GetPlayers()
	for i, player in pls do
		local character = player.Character
		local humanoid = character:WaitForChild("Humanoid")
		--humanoid health equals 0
		--use a method from the humanoid documentation for the humanoid to take damage
		humanoid:TakeDamage(90000)
	end
end

local function DeleteMap()
	game.Workspace.GameMap:ClearAllChildren()
end

while true do
	if #Players:GetPlayers() >= MinPlayers then
		for count = Intermission, 0, -1 do
			print(count)
			task.wait(1)
		end
		AssignTeams()
		ChooseMap()
		TpToGame()
		GiveWeapon()
		runGame()
		ClearWeapons()
		ClearTeams()
		TpToLobby()
		DeleteMap()
	end
	task.wait(1)
	print("Not enough players")
end
